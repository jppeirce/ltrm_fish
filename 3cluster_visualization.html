We next inspected how each cluster varied in it’s species composition.

We will produce three vertical heatmaps for the Jaccard clustering method. Each heatmap shows the average percent presence of each fish species (rows) within each cluster (columns).

What the graph means:

  • Columns: Each column is a cluster of assemblages (sites grouped by fish community similarity).
  • Rows: Each row is a fish species (only those present in >10% of any cluster).
  • Cell values: The number in each cell is the percent of assemblages in that cluster where the species is present (0–100%).
  • Color: Darker blue = higher percent presence.
  • Interpretation: You can see which species are common or rare in each cluster, and how species composition differs between clusters and clustering methods.

import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
from scipy import optimize
from scipy.stats import f as f_dist
from plotnine import *

# =====================================================================
# SEGMENTED REGRESSION FOR ALL POOL-CLUSTER COMBINATIONS
# =====================================================================

def piecewise_linear(x, x0, y0, m1, m2):
    """
    Continuous piecewise linear function with breakpoint at x0.
    Left  (x <= x0): y = y0 + m1*(x - x0)
    Right (x >  x0): y = y0 + m2*(x - x0)

    Written with np.maximum instead of np.piecewise so the function is
    differentiable w.r.t. x0 -- curve_fit's numerical Jacobian is reliable.
    """
    return y0 + m1 * (x - x0) + (m2 - m1) * np.maximum(0.0, x - x0)


def fit_segmented_regression(years, proportions, min_points=6, min_nonzero=4, edge_frac=0.15):
    """
    Fit continuous segmented regression with one breakpoint using curve_fit.

    Parameters
    ----------
    min_points : int
        Minimum number of yearly observations required. With zero-filled
        data every cluster in a pool has the same number of points (one per
        sampled pool-year), so this effectively requires the pool to have
        been sampled in >= min_points distinct years.
    min_nonzero : int
        Minimum number of nonzero proportions required. Zero-filling creates
        many near-all-zero series for rare clusters; fitting a 4-parameter
        piecewise model to those is degenerate, so skip them.

    Returns
    -------
    dict with fit results or None if insufficient data
    """
    years = np.asarray(years, dtype=float)
    proportions = np.asarray(proportions, dtype=float)

    if len(years) < min_points:
        return None

    # Guard against degenerate series introduced by zero-filling:
    # too few nonzero points, or an essentially constant series.
    if np.count_nonzero(proportions) < min_nonzero:
        return None
    if np.ptp(proportions) == 0:
        return None

    yr_min, yr_max = years.min(), years.max()
    yr_range = yr_max - yr_min
    if yr_range <= 0:
        return None

    # Keep the breakpoint away from the data extremes so each segment
    # retains a meaningful number of points.
    buf = yr_range * edge_frac
    lo_x0, hi_x0 = yr_min + buf, yr_max - buf

    # Initial guesses
    x0_guess = float(np.clip(np.median(years), lo_x0, hi_x0))
    y0_guess = proportions[np.argmin(np.abs(years - x0_guess))]

    # Initialize slopes from the two data halves so the optimizer has a
    # non-zero gradient signal to move the breakpoint.
    mid = int(np.searchsorted(years, x0_guess))
    if 1 < mid < len(years) - 1:
        m1_guess = np.polyfit(years[:mid], proportions[:mid], 1)[0]
        m2_guess = np.polyfit(years[mid:], proportions[mid:], 1)[0]
    else:
        m1_guess = m2_guess = np.polyfit(years, proportions, 1)[0]

    p0 = [x0_guess, y0_guess, m1_guess, m2_guess]

    # Bounds: breakpoint constrained to the buffered interior range
    bounds = (
        [lo_x0, -np.inf, -np.inf, -np.inf],
        [hi_x0,  np.inf,  np.inf,  np.inf]
    )

    try:
        # Fit piecewise model
        params, cov = optimize.curve_fit(
            piecewise_linear,
            years, proportions,
            p0=p0,
            bounds=bounds,
            maxfev=10000
        )

        x0_fit, y0_fit, m1_fit, m2_fit = params
        y_hat = piecewise_linear(years, *params)
        rss_piece = np.sum((proportions - y_hat) ** 2)

        # Simple linear model for comparison
        slope_simple, intercept_simple = np.polyfit(years, proportions, 1)
        y_hat_simple = slope_simple * years + intercept_simple
        rss_simple = np.sum((proportions - y_hat_simple) ** 2)

        # F-test (nested-model comparison).
        # NOTE: because the breakpoint x0 is a free parameter that is
        # unidentified under the null, this p-value is anti-conservative
        # (Davies problem). Treat it as a screening heuristic, not an
        # exact significance test.
        n = len(years)
        k_piece = 4   # x0, y0, m1, m2
        k_simple = 2

        if n > k_piece and rss_piece > 0 and rss_simple > rss_piece:
            f_stat = ((rss_simple - rss_piece) / (k_piece - k_simple)) / (rss_piece / (n - k_piece))
            p_value = 1 - f_dist.cdf(f_stat, k_piece - k_simple, n - k_piece)
        else:
            # Segmented fit did not improve on simple linear (or too few
            # points) -- F-test is not meaningful.
            f_stat = np.nan
            p_value = np.nan

        return {
            'breakpoint': x0_fit,
            'y_at_break': y0_fit,
            'slope_before': m1_fit,
            'slope_after': m2_fit,
            'rss_piecewise': rss_piece,
            'rss_simple': rss_simple,
            'improvement_pct': 100 * (rss_simple - rss_piece) / rss_simple if rss_simple > 0 else 0,
            'f_stat': f_stat,
            'p_value': p_value,
            'significant': (p_value < 0.05) if not np.isnan(p_value) else False,
            'n_points': n,
            'params': params,
            'cov': cov
        }

    except Exception as e:
        print(f"  Warning: Fit failed - {str(e)}")
        return None

# Without this, a cluster absent in some years would be averaged only over
# the years it appeared in — biasing the mean upward.
all_combos = pd.MultiIndex.from_product(
    [annual_counts['year'].unique(),
     annual_counts['mode_aqua_desc'].unique(),
     highlight_clusters,
     ['Before breakpoint', 'After breakpoint']],
    names=['year', 'mode_aqua_desc', 'cluster', 'period']
).to_frame(index=False)
annual_counts = all_combos.merge(annual_counts, how='left').fillna({'count': 0})

# Step 3: within-year proportions (denominator = total samples that year/pool/period)
annual_totals = annual_counts.groupby(['year', 'mode_aqua_desc', 'period'])['count'].transform('sum')
annual_counts['proportion'] = annual_counts['count'] / annual_totals
annual_counts = annual_counts.dropna(subset=['proportion'])  # drops year/pool/period with 0 total

# Step 4: mean across years
prop_df = (annual_counts.groupby(['mode_aqua_desc', 'cluster', 'period'])['proportion']
           .mean()
           .reset_index())

# Categorical ordering
prop_df['cluster']   = pd.Categorical(prop_df['cluster'],   categories=highlight_clusters, ordered=True)
prop_df['mode_aqua_desc'] = pd.Categorical(prop_df['mode_aqua_desc'], categories=annual_counts['mode_aqua_desc'].unique(), ordered=True)
prop_df['period']    = pd.Categorical(prop_df['period'],
                                       categories=['Before breakpoint', 'After breakpoint'],
                                       ordered=True)

# Create faceted plot
g = sns.FacetGrid(prop_df, col='period', height=5, aspect=1.2, sharey=True)
g.map_dataframe(sns.barplot, x='mode_aqua_desc', y='proportion', hue='cluster',
                palette=cluster_colors, dodge=True, order=annual_counts['mode_aqua_desc'].unique(),
                hue_order=highlight_clusters)

g.set_axis_labels('Pool', 'Mean Annual Proportion of Samples')
g.set_titles(col_template='{col_name}')
g.add_legend(title='Cluster', bbox_to_anchor=(1.02, 0.5), loc='center left')

for ax in g.axes.flat:
    ax.tick_params(axis='y', left=True, labelleft=True)
    ax.tick_params(axis='x', rotation=45)
    for label in ax.get_xticklabels():
        label.set_ha('right')

plt.tight_layout()
plt.savefig('cluster_aqua_desc_proportion_comparison.jpg', dpi=300, bbox_inches='tight',
            format='jpeg', pil_kwargs={'quality': 95})
plt.show()
